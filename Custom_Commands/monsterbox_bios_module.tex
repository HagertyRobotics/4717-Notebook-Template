\newcommand{\dndbio}[7]{
	\newpage
	\addcontentsline{toc}{section}{\numberline{}#1}
	\begin{figure}
	\centering
	\includegraphics[height=.45\paperwidth, keepaspectratio]{Bios/Images/#2}
	\end{figure}
	\begin{monsterboxnobg}{#1}
		\begin{hangingpar}
	    	\textit{``#3''}
	  	\end{hangingpar}
	\dndline
	\basics[%
		committees = \Large{#4},
		Current Grade = \Large{#5},
		Stats = \Large{#6}
	]
	\vspace{3mm}
	\stats[
	    STAT = \stat{#7}
	]
	\dndline%
	\end{monsterboxnobg}
}
